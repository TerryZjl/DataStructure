void CountSort(int *a, size_t size)
{
	int min = a[0];
	int max = a[0];
	for (int i = 1; i < size; i++)
	{
		if (min > a[i])
			min = a[i];
		if (max < a[i])
			max = a[i];
	}
	int range = max - min + 1;
	vector<int> tmp;
	tmp.resize(range, 0);
	for (int i = 0; i < size; i++)
	{
		int index = a[i] % range;
		tmp[index]++;
	}
	for (size_t i = 0,j = 0; i < range; i++)
	{
		while (tmp[i]-- != 0)
			a[j++] = i+min;
	}
}
