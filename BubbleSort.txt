void BubbleSort(int *a, size_t n)
{
	for (size_t j = n-1; j > 0; j--)//无序区间逐渐变小
	{
		bool IsSequence = true;
		for (size_t i = 0; i < j; i++)//每一趟从无序区间里冒出来一个最大的
		{
			if (a[i]> a[i + 1])
			{
				IsSequence = false;
				swap(a[i], a[i + 1]);//每比较进来一次就交换一次，所以是交换排序，选择排序每次进来只选出坐标。
			}
		}
		if (IsSequence)
			break;
	}
}
